package timewheel

import (
	"database/sql"
	"encoding/json"
	"fmt"
	"sync"
	"time"

	_ "modernc.org/sqlite"
)

// SQLiteTaskStore SQLite 任务存储
type SQLiteTaskStore struct {
	db   *sql.DB
	once sync.Once
}

// SQLiteHistoryStore SQLite 历史存储
type SQLiteHistoryStore struct {
	db       *sql.DB
	recordCh chan AlertHistory
	stopCh   chan struct{}
	wg       sync.WaitGroup
}

const (
	schemaAlertRules = `
CREATE TABLE IF NOT EXISTS alert_rules (
    id TEXT PRIMARY KEY,
    mode INTEGER DEFAULT 2,
    interval_ms INTEGER NOT NULL,
    priority INTEGER DEFAULT 1,
    severity INTEGER DEFAULT 1,
    for_duration_ms INTEGER DEFAULT 0,
    repeat_interval_ms INTEGER DEFAULT 0,
    times INTEGER DEFAULT 0,
    timeout_ms INTEGER DEFAULT 0,
    description TEXT,
    labels TEXT,
    annotations TEXT,
    enabled INTEGER DEFAULT 1,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);`

	schemaAlertHistory = `
CREATE TABLE IF NOT EXISTS alert_history (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    task_id TEXT NOT NULL,
    old_state INTEGER NOT NULL,
    new_state INTEGER NOT NULL,
    value REAL,
    threshold REAL,
    is_firing INTEGER DEFAULT 0,
    severity INTEGER,
    labels TEXT,
    annotations TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);`

	indexAlertRulesEnabled   = `CREATE INDEX IF NOT EXISTS idx_alert_rules_enabled ON alert_rules(enabled);`
	indexAlertHistoryTaskID  = `CREATE INDEX IF NOT EXISTS idx_alert_history_task_id ON alert_history(task_id);`
	indexAlertHistoryCreated = `CREATE INDEX IF NOT EXISTS idx_alert_history_created_at ON alert_history(created_at);`
)

// NewSQLiteStore 创建 SQLite 存储（同时用于任务和历史）
func NewSQLiteStore(dbPath string) (*SQLiteTaskStore, *SQLiteHistoryStore, error) {
	db, err := sql.Open("sqlite", dbPath)
	if err != nil {
		return nil, nil, fmt.Errorf("failed to open database: %w", err)
	}

	// 创建表
	if _, err := db.Exec(schemaAlertRules); err != nil {
		db.Close()
		return nil, nil, fmt.Errorf("failed to create alert_rules table: %w", err)
	}
	if _, err := db.Exec(schemaAlertHistory); err != nil {
		db.Close()
		return nil, nil, fmt.Errorf("failed to create alert_history table: %w", err)
	}
	if _, err := db.Exec(indexAlertRulesEnabled); err != nil {
		db.Close()
		return nil, nil, fmt.Errorf("failed to create index: %w", err)
	}
	if _, err := db.Exec(indexAlertHistoryTaskID); err != nil {
		db.Close()
		return nil, nil, fmt.Errorf("failed to create index: %w", err)
	}
	if _, err := db.Exec(indexAlertHistoryCreated); err != nil {
		db.Close()
		return nil, nil, fmt.Errorf("failed to create index: %w", err)
	}

	taskStore := &SQLiteTaskStore{db: db}
	historyStore := &SQLiteHistoryStore{
		db:       db,
		recordCh: make(chan AlertHistory, 1000),
		stopCh:   make(chan struct{}),
	}

	// 启动异步写入器
	historyStore.startBatchWriter()

	return taskStore, historyStore, nil
}

// Save 保存任务
func (s *SQLiteTaskStore) Save(task *Task) error {
	if task == nil {
		return ErrInvalidParam
	}

	labelsJSON, _ := json.Marshal(task.Labels)
	annotationsJSON, _ := json.Marshal(task.Annotations)

	query := `
INSERT OR REPLACE INTO alert_rules
(id, mode, interval_ms, priority, severity, for_duration_ms, repeat_interval_ms, times, timeout_ms, description, labels, annotations, enabled, updated_at)
VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, 1, CURRENT_TIMESTAMP)`

	_, err := s.db.Exec(query,
		task.ID,
		task.Mode,
		task.Interval.Milliseconds(),
		task.Priority,
		task.Severity,
		task.For.Milliseconds(),
		task.RepeatInterval.Milliseconds(),
		task.Times,
		task.Timeout.Milliseconds(),
		task.Description,
		string(labelsJSON),
		string(annotationsJSON),
	)
	return err
}

// Delete 删除任务
func (s *SQLiteTaskStore) Delete(taskID string) error {
	_, err := s.db.Exec("DELETE FROM alert_rules WHERE id = ?", taskID)
	return err
}

// LoadAll 加载所有任务
func (s *SQLiteTaskStore) LoadAll() ([]*Task, error) {
	return s.loadTasks(false)
}

// LoadEnabled 只加载启用的任务
func (s *SQLiteTaskStore) LoadEnabled() ([]*Task, error) {
	return s.loadTasks(true)
}

func (s *SQLiteTaskStore) loadTasks(enabledOnly bool) ([]*Task, error) {
	query := "SELECT id, mode, interval_ms, priority, severity, for_duration_ms, repeat_interval_ms, times, timeout_ms, description, labels, annotations FROM alert_rules"
	if enabledOnly {
		query += " WHERE enabled = 1"
	}

	rows, err := s.db.Query(query)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	var tasks []*Task
	for rows.Next() {
		task := &Task{}
		var labelsJSON, annotationsJSON string
		var intervalMs, forMs, repeatMs, timeoutMs int64

		err := rows.Scan(
			&task.ID,
			&task.Mode,
			&intervalMs,
			&task.Priority,
			&task.Severity,
			&forMs,
			&repeatMs,
			&task.Times,
			&timeoutMs,
			&task.Description,
			&labelsJSON,
			&annotationsJSON,
		)
		if err != nil {
			return nil, err
		}

		task.Interval = time.Duration(intervalMs) * time.Millisecond
		task.For = time.Duration(forMs) * time.Millisecond
		task.RepeatInterval = time.Duration(repeatMs) * time.Millisecond
		task.Timeout = time.Duration(timeoutMs) * time.Millisecond

		if labelsJSON != "" {
			json.Unmarshal([]byte(labelsJSON), &task.Labels)
		}
		if annotationsJSON != "" {
			json.Unmarshal([]byte(annotationsJSON), &task.Annotations)
		}

		tasks = append(tasks, task)
	}

	return tasks, rows.Err()
}

// Close 关闭存储
func (s *SQLiteTaskStore) Close() error {
	if s.db != nil {
		return s.db.Close()
	}
	return nil
}

// Record 异步记录历史（非阻塞）
func (s *SQLiteHistoryStore) Record(record AlertHistory) error {
	select {
	case s.recordCh <- record:
		return nil
	default:
		return fmt.Errorf("history record channel full")
	}
}

// GetHistory 获取历史记录
func (s *SQLiteHistoryStore) GetHistory(taskID string, limit int) ([]AlertHistory, error) {
	var query string
	var args []interface{}

	if taskID != "" {
		query = "SELECT task_id, old_state, new_state, value, threshold, is_firing, severity, labels, annotations, created_at FROM alert_history WHERE task_id = ? ORDER BY created_at DESC"
		args = []interface{}{taskID}
	} else {
		query = "SELECT task_id, old_state, new_state, value, threshold, is_firing, severity, labels, annotations, created_at FROM alert_history ORDER BY created_at DESC"
		args = []interface{}{}
	}

	if limit > 0 {
		query += fmt.Sprintf(" LIMIT %d", limit)
	}

	rows, err := s.db.Query(query, args...)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	var history []AlertHistory
	for rows.Next() {
		var h AlertHistory
		var labelsJSON, annotationsJSON string
		var isFiring int

		err := rows.Scan(
			&h.TaskID,
			&h.OldState,
			&h.State,
			&h.Value,
			&h.Threshold,
			&isFiring,
			&h.Severity,
			&labelsJSON,
			&annotationsJSON,
			&h.Timestamp,
		)
		if err != nil {
			return nil, err
		}

		h.IsFiring = isFiring != 0
		if labelsJSON != "" {
			json.Unmarshal([]byte(labelsJSON), &h.Labels)
		}
		if annotationsJSON != "" {
			json.Unmarshal([]byte(annotationsJSON), &h.Annotations)
		}

		history = append(history, h)
	}

	return history, rows.Err()
}

// DeleteOlderThan 删除指定天数之前的记录
func (s *SQLiteHistoryStore) DeleteOlderThan(days int) error {
	cutoff := time.Now().AddDate(0, 0, -days)
	_, err := s.db.Exec("DELETE FROM alert_history WHERE created_at < ?", cutoff)
	return err
}

// Close 关闭存储
func (s *SQLiteHistoryStore) Close() error {
	close(s.stopCh)
	s.wg.Wait()
	if s.db != nil {
		return s.db.Close()
	}
	return nil
}

// startBatchWriter 启动批量写入器
func (s *SQLiteHistoryStore) startBatchWriter() {
	s.wg.Add(1)
	go s.runBatchWriter()
}

func (s *SQLiteHistoryStore) runBatchWriter() {
	defer s.wg.Done()

	batch := make([]AlertHistory, 0, 100)
	ticker := time.NewTicker(5 * time.Second)
	defer ticker.Stop()

	flush := func() {
		if len(batch) == 0 {
			return
		}

		tx, err := s.db.Begin()
		if err != nil {
			return
		}

		stmt, err := tx.Prepare(`
INSERT INTO alert_history (task_id, old_state, new_state, value, threshold, is_firing, severity, labels, annotations, created_at)
VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`)
		if err != nil {
			tx.Rollback()
			return
		}
		defer stmt.Close()

		for _, h := range batch {
			labelsJSON, _ := json.Marshal(h.Labels)
			annotationsJSON, _ := json.Marshal(h.Annotations)
			isFiring := 0
			if h.IsFiring {
				isFiring = 1
			}

			stmt.Exec(h.TaskID, h.OldState, h.State, h.Value, h.Threshold, isFiring, h.Severity, string(labelsJSON), string(annotationsJSON), h.Timestamp)
		}

		tx.Commit()
		batch = batch[:0]
	}

	for {
		select {
		case record := <-s.recordCh:
			batch = append(batch, record)
			if len(batch) >= 100 {
				flush()
			}
		case <-ticker.C:
			flush()
		case <-s.stopCh:
			flush()
			return
		}
	}
}

func (tw *TimeWheel) handleTaskSchedulingLocked(node *taskSlot, taskInfo *taskInfo, now time.Time, currentSlot int) {
	// 移除 shard.mu.Lock() 和 defer shard.mu.Unlock()

// Task 定时任务定义
type Task struct {
	ID             string            // 任务ID（唯一标识）
	Mode           TaskMode          // 执行模式
	Interval       time.Duration     // 执行间隔
	Priority       TaskPriority      // 优先级
	Severity       Severity          // 告警级别
	For            time.Duration     // 持续时间（For）
	RepeatInterval time.Duration     // 重复告警间隔
	Times          int               // 固定次数模式下的执行次数
	Timeout        time.Duration     // 任务超时时间
	Description    string            // 任务描述
	Labels         map[string]string // 标签
	Annotations    map[string]string // 注解
	Run            func(ctx context.Context) AlarmResult // 任务执行函数
}

// AlarmResult 告警结果
type AlarmResult struct {
	Value     float64 // 当前值
	Threshold float64 // 阈值
	IsFiring  bool    // 是否触发
}

// AlertHistory 告警历史记录
type AlertHistory struct {

	IsFiring    bool       `json:"is_firing"`
	Severity    Severity   `json:"severity"`

}

	}

		// 可以添加更多检查
	}

	return &HealthStatus{
		Status:     status,
		Running:    tw.running.Load(),
		TaskCount:  tw.Stats().TotalTasks,
		Uptime:     time.Since(tw.startTime).String(),
		StartTime:  tw.startTime,
		Components: components,
	}
}

// HTTPHandler 返回 HTTP 处理器
func (tw *TimeWheel) HTTPHandler() http.Handler {
	mux := http.NewServeMux()

	// 健康检查
	mux.HandleFunc("/health", func(w http.ResponseWriter, r *http.Request) {
		health := tw.Health()
		w.Header().Set("Content-Type", "application/json")
		if health.Status != "healthy" {
			w.WriteHeader(http.StatusServiceUnavailable)
		}
		json.NewEncoder(w).Encode(health)
	})

	// 就绪检查
	mux.HandleFunc("/ready", func(w http.ResponseWriter, r *http.Request) {
		w.Header().Set("Content-Type", "application/json")
		if !tw.running.Load() {
			w.WriteHeader(http.StatusServiceUnavailable)
			json.NewEncoder(w).Encode(map[string]bool{"ready": false})
			return
		}
		json.NewEncoder(w).Encode(map[string]bool{"ready": true})
	})

	return mux
}

func (tw *TimeWheel) Start() error {
	// ... 现有启动逻辑 ...

	// 自动恢复任务
	if tw.autoRestore && tw.taskStore != nil {
		tasks, err := tw.taskStore.LoadEnabled()
		if err != nil {
			tw.logger.Printf("[WARN] 加载任务失败: %v", err)
		} else {
			for _, task := range tasks {
				if err := tw.AddTask(task); err != nil {
					tw.logger.Printf("[WARN] 恢复任务失败: id=%s, err=%v", task.ID, err)
				} else {
					tw.logger.Printf("[INFO] 恢复任务成功: id=%s", task.ID)
				}
			}
		}
	}

	return nil
}
